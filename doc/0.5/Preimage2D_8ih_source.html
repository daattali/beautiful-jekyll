00456 void
00457 DGtal::Preimage2D<Shape>::getSeparatingStraightLine(
00458   double& alpha, 
00459   double& beta, 
00460   double& gamma) const
00461 {

00463   Point Uf = getUf();
00464   Point Ul = getUl();
00465   Point Lf = getLf();
00466   Point Ll = getLl();
00467 
00468   //parameters
00469   typedef typename Point::Coordinate Coordinate; 
00470   
00471   double a = -NumberTraits<Coordinate>::castToDouble(Ll[1]-Uf[1]);
00472   double b = NumberTraits<Coordinate>::castToDouble(Ll[0]-Uf[0]);
00473   double c = -NumberTraits<Coordinate>::castToDouble(Uf[0])*a 
00474                      -NumberTraits<Coordinate>::castToDouble(Uf[1])*b;
00475   
00476   double ap = -NumberTraits<Coordinate>::castToDouble(Ul[1]-Lf[1]);
00477   double bp = NumberTraits<Coordinate>::castToDouble(Ul[0]-Lf[0]);
00478   double cp = -NumberTraits<Coordinate>::castToDouble(Lf[0])*ap 
00479                       -NumberTraits<Coordinate>::castToDouble(Lf[1])*bp;
00480 
00481   double x, y; 
00482   double det = (ap*b-bp*a); 
00483   if (det != 0)
00484   {
00485     //intersection point
00486     x = -(b*cp-bp*c)/det;
00487     y = -(a*cp-ap*c)/(-det); 
00488     //normalisation of (ap,bp) with respect to (a,b)
00489     double l = std::sqrt(a*a + b*b); 
00490     double lp = std::sqrt(ap*ap + bp*bp); 
00491     double apn = ap/lp*l; 
00492     double bpn = bp/lp*l; 
00493     //slope bisector of (UfLl) and (UlLf)
00494     alpha = (a+apn)/2.0; 
00495     beta = (b+bpn)/2.0; 
00496     //intercept
00497     gamma = -alpha*x - beta*y;
00498   }
00499   else 
00500   {//parallel case
00501     alpha = a;
00502     beta = b;
00503     gamma = c;
00504   }
