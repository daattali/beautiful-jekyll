00051 template <typename Quotient>
00052 bool
00053 equalCFrac( const std::vector<Quotient> & c1, const std::vector<Quotient> & c2 )
00054 {
00055   unsigned int s = c1.size() < c2.size() ? c1.size() : c2.size();
00056   if ( ( s != c1.size() ) && ( c1.back() != NumberTraits<Quotient>::ONE ) )
00057     return false;
00058   if ( ( s != c2.size() ) && ( c2.back() != NumberTraits<Quotient>::ONE ) )
00059     return false;
00060   for ( unsigned int i = 0; i < s; ++i )
00061     {
00062       Quotient q1 = c1[ i ];
00063       if ( ( s != c1.size() ) && ( i == s - 1 ) ) q1 += c1.back();
00064       Quotient q2 = c2[ i ];
00065       if ( ( s != c2.size() ) && ( i == s - 1 ) ) q2 += c2.back();
00066       if ( q1 != q2 ) return false;
00067     }
00068   return true;
00069 }
