172  double alpha1 = std::atan2( NumberTraits<Coordinate>::castToDouble(aPoint1[1])-cy,
173  NumberTraits<Coordinate>::castToDouble(aPoint1[0])-cx );
174  alpha1 = (alpha1<0)?(alpha1+2.0*M_PI):alpha1;
175  double alpha2 = std::atan2( NumberTraits<Coordinate>::castToDouble(aPoint2[1])-cy,

184  }
185  catch ( InfiniteNumberException e )
186  {
187  aBoard.drawLine( NumberTraits<Coordinate>::castToDouble(aPoint1[0]),
188  NumberTraits<Coordinate>::castToDouble(aPoint1[1]),

500

502  ConstIterator it (g.end());
503  --it;

508  double ly2 = NumberTraits<Coordinate>::castToDouble(lastPair.second[1]);

548  double ly2 = NumberTraits<Coordinate>::castToDouble(lastPair.second[1]);

572  else if ( mode == "Annulus" )
