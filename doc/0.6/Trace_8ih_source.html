217 inline void
218 DGtal::Trace::progressBar(const double currentValue, const double maximumValue)
219 {
220  // how wide you want the progress meter to be
221  double fraction = currentValue /maximumValue;
222 
223  // part of the progressmeter that's already "full"
224  int dotz = (int)floor(fraction * PROGRESSBARWIDTH);
225  if (dotz == PROGRESSBARWIDTH)
226  return;
227 
228  myProgressBarCurrent = dotz;
229 
230  // create the "meter"
231  int ii=0;
232  myWriter.outputStream() << myCurrentPrefix<<"[";
233  // part that's full already
234  for ( ; ii < dotz;ii++)
235  {
236  myWriter.outputStream()<< "#";
237  }
238  // remaining part (spaces)
239  for ( ; ii < PROGRESSBARWIDTH;ii++)
240  {
241  myWriter.outputStream()<< " ";
242  }
243  // and back to line begin - do not forget the fflush to avoid output buffering problems!
244  myWriter.outputStream()<< "] "<< (int)(fraction*100)<<"/100\r";
245  myWriter.outputStream().flush();
246 }
