57 {
58 
59  trace.info() << endl;

62  typedef typename IteratorCirculatorTraits<typename Range::ConstIterator>::Value Value;

66  trace.info() << "Forward" << endl;
67  typename Range::ConstIterator i = aRange.begin();
68  typename Range::ConstIterator end = aRange.end();
69  for ( ; i != end; ++i) {
70  cout << *i << " ";

72  }
73  cout << endl;
74  }
75  {
76  trace.info() << "Backward" << endl;
77  typename Range::ConstReverseIterator i = aRange.rbegin();
78  typename Range::ConstReverseIterator end = aRange.rend();
79  for ( ; i != end; ++i) {
80  cout << *i << " ";
81  v2.push_back(*i);
82  }
83  cout << endl;
84  }
85  {
86  trace.info() << "Circulator" << endl;
87  typename Range::ConstCirculator c = aRange.c();
88  typename Range::ConstCirculator cend = aRange.c();
89  if (isNotEmpty(c,cend))
90  {
91  do
92  {
93  cout << *c << " ";
94  l3.push_back(*c);
95  c++;
96  } while (c!=cend);
97  }
98  cout << endl;
99  }
100 
101  {
102  trace.info() << "Reverse Circulator" << endl;
103  typename Range::ConstReverseCirculator c = aRange.rc();
104  typename Range::ConstReverseCirculator cend = aRange.rc();
105  if (isNotEmpty(c,cend))
106  {
107  do

109  cout << *c << " ";
110  l4.push_back(*c);
111  c++;
112  } while (c!=cend);
113  }
114  cout << endl;
115  }
116 
117  return ( std::equal(v1.begin(),v1.end(),l3.begin())
118  && std::equal(v1.begin(),v1.end(),v2.rbegin())
