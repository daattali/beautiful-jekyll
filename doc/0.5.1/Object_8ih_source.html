464 {

467  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
468  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
469  //typedef typename Domain::Predicate Predicate;
470 
471  // Intermediate container that is fast writable.
472  Container neigh;
473  const DigitalSet & mySet = pointSet();
474  Object<DigitalTopology, DigitalSet> output( topology(),
475  domain() );
476  // mySet.domain() );
477  DigitalSet & outputSet = output.pointSet();
478 
479  // Loop on all points of the set.
480  const DigitalSetConstIterator it_end = mySet.end();
481  for ( DigitalSetConstIterator it = mySet.begin();
482  it != it_end;
483  ++it )
484  {
485  std::back_insert_iterator< Container > back_ins_it( neigh );
486  // Computing neighborhood within domain.

513 template <typename TDigitalTopology, typename TDigitalSet>
514 template <typename OutputObjectIterator>
515 inline
516 typename DGtal::Object<TDigitalTopology, TDigitalSet>::Size
517 DGtal::Object<TDigitalTopology, TDigitalSet>
518 ::writeComponents( OutputObjectIterator & it ) const
519 {
520  Size nb_components = 0;
521  if ( pointSet().empty() )
522  {
523  myConnectedness = CONNECTED;
524  return nb_components;
525  }
526  else
527  if ( connectedness() == CONNECTED )
528  {
529  *it++ = *this;
530  return 1;
531  }
532  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
533  DigitalSetConstIterator it_object = pointSet().begin();
534  Point p( *it_object++ );
535  Expander<Object> expander( *this, p );
536  while ( expander.nextLayer() )
537  ;
538  Object component( myTopo, expander.core(), CONNECTED );
539  *it++ = component;
540  ++nb_components;
541  DigitalSet visited( expander.core() );
542  while ( it_object != pointSet().end() )
543  {
544  p = *it_object++;
545  if ( visited.find( p ) == visited.end() )
546  {
547  Expander<Object> expander2( *this, p );
548  while ( expander2.nextLayer() )
549  ;
550  Object component2( myTopo, expander2.core(), CONNECTED );
551  *it++ = component2;
552  ++nb_components;
553  visited += expander2.core();
554  }
555  }
556  myConnectedness = nb_components == 1 ? CONNECTED : DISCONNECTED;
557  return nb_components;
558 }

584 DGtal::Connectedness

614 template <typename TDigitalTopology, typename TDigitalSet>
615 template <typename TAdjacency>
616 inline
617 typename DGtal::Object<TDigitalTopology, TDigitalSet>::SmallObject
618 DGtal::Object<TDigitalTopology, TDigitalSet>
619 ::geodesicNeighborhood
620 ( const TAdjacency & adj, const Point & p, unsigned int k ) const
621 {
622  // Local types.
623  typedef std::vector<Point> Container;
624  typedef MetricAdjacency<Space, Space::dimension> AlphaAdjacency;
625  typedef MetricAdjacency<Space, 1> OmegaAdjacency;
626  typedef DGtal::DigitalTopology<TAdjacency, OmegaAdjacency> LocalTopology;
627  typedef typename Container::const_iterator ContainerConstIterator;
628  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
629  typedef Object<LocalTopology, SmallSet> LocalObject;
630  typedef HyperRectDomain<Space> LocalDomain;
631 
632  DigitalSetDomain<DigitalSet> limitedX( *myPointSet );
633  AlphaAdjacency alpha;
634  OmegaAdjacency omega;
635  // Intermediate container that is fast writable.
636  Container local_points;
637  std::back_insert_iterator< Container > back_ins_it( local_points );
638  alpha.writeProperNeighborhood( p, back_ins_it, limitedX.predicate() );
639 
640  // Construct local domain.
641  Point p1( p );
642  for ( typename Point::Iterator it = p1.begin(); it != p1.end(); ++it )
643  --(*it);
644  Point p2( p );
645  for ( typename Point::Iterator it = p2.begin(); it != p2.end(); ++it )
646  ++(*it);
647  LocalDomain aDomain( p1, p2 );
648 
649  // Construct local X.
650  LocalTopology aTopology( adj, omega );
651  LocalObject X( aTopology, aDomain );
652  X.pointSet().insertNew( local_points.begin(), local_points.end() );
653 
654  // A neighborhood is small, so is defined the digital object.
655  typename LocalObject::SmallObject neighAdj = X.properNeighborhood( p );
656  Expander<LocalObject> expander( X,
657  neighAdj.pointSet().begin(),
658  neighAdj.pointSet().end() );
659  for ( unsigned int i = 1; ( i < k ) && ( ! expander.finished() ); ++i )
660  expander.nextLayer();
661 
662  SmallObject geodesicN( this->topology(), expander.core() );
663  return geodesicN;
664 
665 }
