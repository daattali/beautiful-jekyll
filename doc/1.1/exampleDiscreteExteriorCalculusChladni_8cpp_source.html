45 double standard_deviation(const VectorXd& xx)
46 {
47  const double mean = xx.mean();
48  double accum = 0;
49  for (int kk=0, kk_max=xx.rows(); kk<kk_max; kk++)
50  accum += (xx(kk)-mean)*(xx(kk)-mean);
51  return sqrt(accum)/(xx.size()-1);
52 }
