312  std::swap(vertices[2], vertices[3]);

318  const Index vertex_index = calculus.getCellIndex(vertex);
319  RealPoint position;
320  for (int dim=0; dim<3; dim++)
321  position[dim] = positions[vertex_index+dim*nvertices];
